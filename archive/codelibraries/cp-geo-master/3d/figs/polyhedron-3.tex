\newcommand{\tri}[4]{%
    \fill[#4, opacity=.9] (#1) -- (#2) -- (#3) -- cycle;
}%
\newcommand{\fig}[2]{%
    \begin{tikzpicture}[tdplot_main_coords, scale=#1]
        #2
    \end{tikzpicture}
}%
\newcommand{\figA}[2]{%
    \fig{#1}{
        \coordinate (A) at (1,0,0);
        \coordinate (B) at (1,1,0);
        \coordinate (C) at (0,1,0);
        \coordinate (D) at (0,0,0);
        #2
    }
}%
\newcommand{\figB}[2]{%
    \fig{#1}{
        \coordinate (A) at (1,0,0);
        \coordinate (B) at (.5,.5,0);
        \coordinate (C) at (1,1,0);
        \coordinate (D) at (0,.5,0);
        #2
    }
}%
\newcommand{\triA}[1]{
    \tri{A}{B}{C}{#1}
    \draw[thin,gray,dotted] (A) -- (B) -- (C) -- (D) -- cycle;
    \draw[-stealth] (A) -- (B);
    \draw[-stealth] (A) -- (C);
}%
\newcommand{\triB}[1]{
    \tri{A}{C}{D}{#1}
    \draw[thin,gray,dotted] (A) -- (B) -- (C) -- (D) -- cycle;
    \draw[-stealth] (A) -- (C);
    \draw[-stealth] (A) -- (D);
}%
