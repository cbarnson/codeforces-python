\newcommand{\fig}[4]{
    \begin{tikzpicture}
    #2
    \tkzClip
    \tkzDefPoints{0/0/A, 2/1/B}
    \tkzDefPointBy[rotation = center A angle #1](B) \tkzGetPoint{C'}
    \tkzDefPointBy[homothety = center A ratio .7](C') \tkzGetPoint{C}
    
    #3
    
    \draw (B) -- (A) -- (C);
    \tkzMarkAngle[size=.5, arrows=->](B,A,C)
    
    \tkzLabelPoints[right](B)
    
    #4
    \end{tikzpicture}
}%
