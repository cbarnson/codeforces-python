\newcommand{\abc}[3]{
    \tkzInit[xmin=-1,xmax=3.5,ymin=-1.5,ymax=1.8]
    \tkzClip
    \tkzDefPoints{#1/A, #2/B, #3/C}
    \tkzDrawPoints(A,B,C)
}%
\newcommand{\circum}{
    \tkzCircumCenter(A,B,C) \tkzGetPoint{O}
    \tkzLabelPoints[left](A)
    \tkzLabelPoints[right](B,C)
    %\tkzDrawPoint(O)
}%
