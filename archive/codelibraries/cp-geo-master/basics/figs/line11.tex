\newcommand{\setup}[2]{
    \tkzInit[xmin=-.6,xmax=4.5,ymin=#2,ymax=3]
    \tkzClip
    \tkzDefPoints{0/0/A, 4/2/B, 1/2/P}
    \tkzDefPointBy[projection = onto A--B](P) \tkzGetPoint{proj}
    \tkzDefPointBy[reflection = over A--B](P) \tkzGetPoint{refl}
    
    \tkzDrawSegment(P,#1)
    \tkzMarkRightAngle(B,proj,P)
    \tkzDrawSegments[->](A,B)
    \tkzDrawPoints(P,#1)
    
    \tkzLabelPoints[above](P)
    \tkzLabelPoint[below](#1){$P'$}
    \tkzLabelSegment[pos=0.8, below](A,B){$l$}
}%
